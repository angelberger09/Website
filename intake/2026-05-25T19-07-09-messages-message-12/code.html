  };

  // Tracking Active Snapped Section using IntersectionObserver
  useEffect(() => {
    const observerOptions = {
      root: containerRef.current,
      threshold: 0.5,
    };

    const observerCallback = (entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          setActiveSection(entry.target.id);
        }
      });
    };

    Object.values(sectionsRef).forEach((ref) => {
      if (ref.current) observer.observe(ref.current);

      Object.values(sectionsRef).forEach((ref) => {
        if (ref.current) observer.unobserve(ref.current);
      });

      setActiveSection(secId);
    }

    if (!sectionsRef.home.current) return;

                        Digital Mail Address

                        type="email" 
                        required
                        value={senderEmail}
                        onChange={(e) => setSenderEmail(e.target.value)}

                      />
                    </div>

          >
            {/* Close */}
            <button 
              onClick={() => setExpandedPolaroid(null)}
              className="absolute -top-10 right-0 text-white/70 hover:text-white flex items-center gap-1 text-[10px] tracking-widest font-sans uppercase"

              <X className="w-3.5 h-3.5" /> CLOSE

            {/* Modal Image */}
            <div className="w-full aspect-square bg-[#EFECE8] overflow-hidden shadow-inner rounded-sm">

                className="w-full h-full object-cover"
              />

            <div className="mt-6 text-center w-full px-2">
              <span className="text-[8px] tracking-widest text-stone-400 font-sans block uppercase mb-1">

              <h3 className="text-stone-800 font-serif text-lg font-bold tracking-tight italic">
                {expandedPolaroid.title}
              </h3>
              
              <div className="h-[1px] w-12 bg-stone-200 mx-auto my-3"></div>

            </div>
          </div>
        </div>
      )}

      {/* ========================================================= */}
      {/* EXPANDABLE MODAL FOR JOURNAL READ */}
      {/* ========================================================= */}
      {selectedJournal && (
        <div

          <div 
            className="bg-[#110d1c] p-6 sm:p-8 rounded-2xl max-w-xl w-full border border-violet-500/20 text-stone-200 relative max-h-[90vh] overflow-y-auto cursor-default scrollbar-none"
            onClick={(e) => e.stopPropagation()}
          >
            <button 
              onClick={() => setSelectedJournal(null)}
              className="absolute top-4 right-4 text-stone-400 hover:text-white flex items-center gap-1 text-[9px] tracking-widest font-sans uppercase"

                src={selectedJournal.image} 
                alt={selectedJournal.title}

            <div className="mt-8 pt-4 border-t border-white/5 flex justify-between text-[9px] font-sans text-stone-500 tracking-wider">

              <span>DATE: {selectedJournal.date}</span>
            </div>
          </div>
        </div>
      )}

      {/* Embedded Animations & Helpers */}
      <style>{`
        @keyframes fadeIn {
          from { opacity: 0; }
          to { opacity: 1; }
        }
        @keyframes slideIn {
          from { transform: translateY(12px); opacity: 0; }

        }
        .animate-fadeIn {
          animation: fadeIn 0.4s ease-out forwards;
        }
        .animate-slideIn {
          animation: slideIn 0.3s cubic-bezier(0.16, 1, 0.3, 1) forwards;
        }
        /* Hide scrollbars cleanly across browsers */
        .scrollbar-none::-webkit-scrollbar {
          display: none;
        }
        .scrollbar-none {
          -ms-overflow-style: none;
          scrollbar-width: none;
        }
      `}</style>

    </div>
  );
