function SpaceBackground() {
  const canvasRef = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    let animationFrameId;
    let stars = [];

    const resizeCanvas = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
      initStars();
    };

    const initStars = () => {
      stars = [];
      const density = Math.floor((canvas.width * canvas.height) / 12000);
      for (let i = 0; i < Math.min(density, 100); i++) {
        stars.push({
          x: Math.random() * canvas.width,
          y: Math.random() * canvas.height,
          radius: Math.random() * 1.0 + 0.3,
          alpha: Math.random() * 0.4 + 0.1,
          speed: Math.random() * 0.04 + 0.01
        });
      }
    };

    const draw = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      for (let star of stars) {
        ctx.fillStyle = `rgba(168, 162, 158, ${star.alpha})`;
        ctx.beginPath();
        ctx.arc(star.x, star.y, star.radius, 0, Math.PI * 2);
        ctx.fill();

        star.x -= star.speed;
        if (star.x < 0) {
          star.x = canvas.width;
          star.y = Math.random() * canvas.height;
        }
      }
      animationFrameId = requestAnimationFrame(draw);
    };

    window.addEventListener('resize', resizeCanvas);
    resizeCanvas();
    draw();

    return () => {
      window.removeEventListener('resize', resizeCanvas);
      cancelAnimationFrame(animationFrameId);
    };
  }, []);

  return (
    <canvas 
      ref={canvasRef} 
      className="fixed inset-0 w-full h-full -z-20 pointer-events-none opacity-[0.3]"
    />
  );
}

    const container = scrollContainerRef.current;
    if (!container) return;

    const observerOptions = {
      root: container,
      threshold: 0.4, // Snaps active indicator when 40% of page is visible
    };

    const observer = new IntersectionObserver((entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          const id = entry.target.id.replace('section-', '');
          setCurrentPage(id);
        }
      });
    }, observerOptions);

                        src={post.image} 
                        alt={post.title} 
                        className="w-full h-full object-cover filter grayscale-[15%] hover:grayscale-0 transition-all duration-500" 
                      />
                    </div>

                    <div className="flex-1 flex flex-col justify-between space-y-3">
                      <div className="space-y-1.5">
                        <div className="flex justify-between items-center text-[9px] font-mono tracking-wider text-stone-400">
                          <span className="text-teal-800 font-semibold">{post.category}</span>
                          <span>{post.date}</span>
                        </div>
                        <h3 className="font-serif text-base text-stone-800 font-medium">
                          {post.title}
                        </h3>
                        <p className="text-xs text-stone-500 font-light leading-relaxed line-clamp-2">
                          {post.excerpt}
                        </p>
                      </div>

                      <div>
                        <button
                          onClick={() => setExpandedBlogId(isExpanded ? null : post.id)}
                          className="px-3 py-1 border border-stone-800 hover:bg-stone-900 hover:text-white transition-colors text-[9px] font-mono tracking-widest uppercase rounded-sm"
                        >
                          {isExpanded ? 'Fold Article' : 'Unfold Thoughts'}
                        </button>
                      </div>
                    </div>
                  </article>
                );
              })}
            </div>

            {/* Stage 3 Animated Content Drawer for active essay */}
            {expandedBlogId && (
              <div className="bg-[#1c1917] text-[#fbf9f6] p-6 rounded-lg border border-stone-800 space-y-3 animate-fade-down">
                <div className="flex justify-between items-center border-b border-stone-800 pb-2">
                  <span className="text-[9px] font-mono text-stone-400 uppercase tracking-widest">
                    Active Essay Reader
                  </span>
                  <button 
                    onClick={() => setExpandedBlogId(null)}
                    className="text-stone-400 hover:text-white"
                  >
                    <X size={14} />
                  </button>
                </div>
                <h4 className="font-serif text-lg italic text-white">
                  {BLOG_POSTS.find(p => p.id === expandedBlogId)?.title}
                </h4>
                <p className="text-xs text-stone-300 font-serif leading-relaxed">
                  {BLOG_POSTS.find(p => p.id === expandedBlogId)?.content}
                </p>
              </div>
            )}

          </div>
        </section>

        {/* ==========================================
            SECTION: GALLERY (Polaroid Tiles First)
            ========================================== */}
        <section 
          id="section-gallery"
          className="snap-start min-h-[calc(100vh-110px)] flex flex-col justify-center px-6 py-12 relative"
        >
          <div className="max-w-xl w-full mx-auto space-y-8">
            
            <div className="text-center space-y-1">
              <span className="text-[9px] tracking-[0.4em] text-stone-400 font-mono uppercase block">Index 02</span>
              <h2 className="text-xl md:text-2xl font-serif text-stone-900 font-normal">The Polaroid Tiles</h2>
              <div className="w-6 h-[1px] bg-stone-400 mx-auto mt-2"></div>
            </div>

            {/* Responsive Polaroid Grid */}
            <div className="grid grid-cols-1 sm:grid-cols-2 gap-6 justify-center">
              {GALLERY_ITEMS.slice(0, 4).map((item, index) => {
                const tilts = ["-rotate-1", "rotate-1", "rotate-1.5", "-rotate-1.5"];
                const curTilt = tilts[index % tilts.length];

                return (
                  <div 
                    key={item.id}
                    onClick={() => setSelectedPolaroid(item)}
                    className={`bg-white/40 backdrop-blur-md p-3 pb-6 shadow-md border border-stone-300/40 rounded-xs transform transition duration-300 ${curTilt} hover:rotate-0 hover:scale-[1.02] cursor-pointer flex flex-col justify-between`}
                  >
                    <div className="aspect-square bg-stone-100 overflow-hidden relative group rounded-xs shadow-inner">
                      <img 
                        src={item.image} 
                        alt={item.title} 
                        className="w-full h-full object-cover filter grayscale-[10%] group-hover:grayscale-0 transition-all duration-300"
                      />
                      <div className="absolute inset-0 bg-stone-900/10 opacity-0 group-hover:opacity-100 transition-opacity flex items-center justify-center">
                        <span className="p-2 bg-white/90 backdrop-blur-sm rounded-full shadow text-stone-800">
                          <Eye size={14} />
                        </span>
                      </div>
                    </div>

                    <div className="pt-3 text-center">
                      <span className="font-serif italic text-xs text-stone-700 tracking-wide block">
                        {item.title}
                      </span>
                      <span className="text-[8px] tracking-widest text-stone-400 font-mono uppercase block mt-1">
                        {item.category}
                      </span>
                    </div>
                  </div>
                );
              })}
            </div>

          </div>
        </section>

        {/* ==========================================
            SECTION: SERVICES (Visual Modules & Simple Interactions)
            ========================================== */}
        <section 
          id="section-services"
          className="snap-start min-h-[calc(100vh-110px)] flex flex-col justify-center px-6 py-12 relative"
        >
          <div className="max-w-xl w-full mx-auto space-y-8">
            
            <div className="text-center space-y-1">
              <span className="text-[9px] tracking-[0.4em] text-stone-400 font-mono uppercase block">Index 03</span>
              <h2 className="text-xl md:text-2xl font-serif text-stone-900 font-normal">Custom Design Commitments</h2>
              <div className="w-6 h-[1px] bg-stone-400 mx-auto mt-2"></div>
            </div>

            {/* Visual Modules */}
            <div className="space-y-4">
              {SERVICES.map((srv) => {
                const isActive = activeService === srv.id;
                return (
                  <div 
                    key={srv.id}
                    className="border border-stone-300/40 bg-white/30 backdrop-blur-md rounded-lg overflow-hidden transition-all duration-300"
                  >
                    <button
                      onClick={() => setActiveService(isActive ? null : srv.id)}
                      className="w-full px-5 py-4 flex items-center justify-between text-left focus:outline-none"
                    >
                      <div>
                        <h3 className="font-serif text-base text-stone-800 font-medium">{srv.title}</h3>
                        <div className="flex gap-3 text-[8px] font-mono text-stone-400 uppercase mt-1">
                          <span className="text-rose-700">{srv.duration}</span>
                          <span>•</span>
                          <span>{srv.price}</span>
                        </div>
                      </div>
                      <ChevronDown 
                        size={16} 
                        className={`text-stone-400 transition-transform duration-300 ${isActive ? 'rotate-180' : ''}`} 
                      />
                    </button>

                    {isActive && (
                      <div className="px-5 pb-5 pt-3 border-t border-stone-300/20 space-y-3 animate-fade-down">
                        <p className="text-stone-500 text-xs font-light leading-relaxed">
                          {srv.description}
                        </p>
                        <div className="space-y-1.5">
                          <span className="text-[8px] uppercase tracking-widest text-stone-400 font-mono block">
                            Direct Deliverables:
                          </span>
                          <ul className="space-y-1">
                            {srv.deliverables.map((item, index) => (
                              <li key={index} className="flex items-center gap-2 text-[11px] text-stone-600">
                                <Check size={10} className="text-stone-600 animate-pulse" />
                                <span>{item}</span>
                              </li>
                            ))}
                          </ul>
                        </div>
                      </div>
                    )}
                  </div>
                );
              })}
            </div>

          </div>
        </section>

        {/* ==========================================
            SECTION: PRODUCTS (Product Cards & Visual Showcase)
            ========================================== */}
        <section 
          id="section-products"
          className="snap-start min-h-[calc(100vh-110px)] flex flex-col justify-center px-6 py-12 relative"
        >
          <div className="max-w-xl w-full mx-auto space-y-8">
            
            <div className="text-center space-y-1">
              <span className="text-[9px] tracking-[0.4em] text-stone-400 font-mono uppercase block">Index 04</span>
              <h2 className="text-xl md:text-2xl font-serif text-stone-900 font-normal">The Atelier Stock</h2>
              <div className="w-6 h-[1px] bg-stone-400 mx-auto mt-2"></div>
            </div>

            {/* Visual Product Grid */}
            <div className="grid grid-cols-2 gap-4">
              {PRODUCTS.map((prod) => (
                <div 
                  key={prod.id}
                  className="bg-white/30 backdrop-blur-md border border-stone-300/40 rounded-lg p-3 flex flex-col justify-between space-y-3 hover:border-stone-400"
                >
                  <div className="aspect-square bg-stone-50 overflow-hidden rounded relative group shadow-inner">
                    <img 
                      src={prod.image} 
                      alt={prod.title} 
                      className="w-full h-full object-cover filter grayscale-[10%]" 
                    />
                    <button 
                      onClick={() => setSelectedProduct(prod)}
                      className="absolute bottom-2 right-2 p-1.5 bg-white/95 rounded-full text-stone-700 shadow-sm"
                      title="Quick Specs"
                    >
                      <Info size={12} />
                    </button>
                  </div>

                  <div className="space-y-1">
                    <h3 className="font-serif text-xs text-stone-800 font-semibold truncate">{prod.title}</h3>
                    <div className="flex justify-between items-center text-[10px] font-mono">
                      <span className="text-stone-400 uppercase">{prod.category}</span>
                      <span className="text-stone-950 font-bold">${prod.price}</span>
                    </div>
                  </div>

                  <button
                    onClick={() => addToCart(prod)}
                    className="w-full py-1.5 bg-stone-900 hover:bg-stone-950 text-[#fbf9f6] text-[9px] font-mono tracking-widest uppercase rounded-sm transition-all"
                  >
                    Acquire Object
                  </button>
                </div>
              ))}
            </div>

          </div>
        </section>

        {/* ==========================================
            SECTION: PROCESS (Creative Process Loop)
            ========================================== */}
        <section 
          id="section-process"
          className="snap-start min-h-[calc(100vh-110px)] flex flex-col justify-center px-6 py-12 relative"
        >
          <div className="max-w-xl w-full mx-auto space-y-8">
            
            <div className="text-center space-y-1">
              <span className="text-[9px] tracking-[0.4em] text-stone-400 font-mono uppercase block">Index 05</span>
              <h2 className="text-xl md:text-2xl font-serif text-stone-900 font-normal">Creative Process</h2>
              <div className="w-6 h-[1px] bg-stone-400 mx-auto mt-2"></div>
            </div>

            {/* Visual Process Timeline Chain */}
            <div className="relative grid grid-cols-1 md:grid-cols-4 gap-4 pt-4">
              {/* Process Cards loop */}
              {PROCESS_STEPS.map((step, idx) => (
                <div 
                  key={step.step}
                  className="bg-white/30 backdrop-blur-md border border-stone-300/40 p-4 rounded-lg relative flex flex-col justify-between space-y-3 hover:scale-[1.01] transition-transform shadow-xs"
                >
                  <div className="flex justify-between items-center">
                    <span className="text-xs font-mono text-stone-400">Step {step.step}</span>
                    <div className="p-1 rounded bg-white border border-stone-200">
                      {step.icon}
                    </div>
                  </div>

                  <div className="space-y-1">
                    <h3 className="font-serif text-sm font-semibold text-stone-800">{step.name}</h3>
                    <p className="text-[10px] text-stone-500 font-light leading-relaxed">
                      {step.excerpt}
                    </p>
                  </div>
                </div>
              ))}
            </div>

            <div className="text-center font-serif italic text-xs text-stone-400 max-w-xs mx-auto">
              \"We reject algorithmic convenience in favor of organic visual friction.\"
            </div>

          </div>
        </section>

        {/* ==========================================
            SECTION: CONTACT (Glass Panel Form)
            ========================================== */}
        <section 
          id="section-contact"
          className="snap-start min-h-[calc(100vh-110px)] flex flex-col justify-center px-6 py-12 relative"
        >
          <div className="max-w-xl w-full mx-auto space-y-8">
            
            <div className="text-center space-y-1">
              <span className="text-[9px] tracking-[0.4em] text-stone-400 font-mono uppercase block">Index 06</span>
              <h2 className="text-xl md:text-2xl font-serif text-stone-900 font-normal">Initiate Correspondence</h2>
              <div className="w-6 h-[1px] bg-stone-400 mx-auto mt-2"></div>
            </div>

            {/* Glass-like Correspondence Box */}
            <div className="bg-white/40 backdrop-blur-md border border-stone-300/40 rounded-xl p-6 shadow-sm">
              {contactSubmitted ? (
                <div className="text-center py-8 space-y-3 animate-fade-down">
                  <div className="w-10 h-10 bg-stone-900 text-[#fbf9f6] rounded-full flex items-center justify-center mx-auto border border-stone-800">
                    <Check size={16} />
                  </div>
                  <h3 className="font-serif text-base text-stone-900">Form Transmitted Successfully</h3>
                  <p className="text-xs text-stone-500 max-w-xs mx-auto leading-relaxed">
                    Letters are reviewed at sunset. You will receive a secure response inside forty-eight hours.
                  </p>
                  <button 
                    onClick={() => setContactSubmitted(false)}
                    className="px-4 py-1.5 border border-stone-400 hover:bg-stone-100 text-[8px] font-mono tracking-widest uppercase rounded-sm"
                  >
                    Transmit another line
                  </button>
                </div>
              ) : (
                <form onSubmit={handleContactSubmit} className="space-y-4">
                  
                  <div className="space-y-1 text-center">
                    <label className="text-[8px] uppercase tracking-widest font-mono text-stone-400">Your Full Name</label>
                    <input 
                      type="text" 
                      required
                      value={contactForm.name}
                      onChange={(e) => setContactForm({ ...contactForm, name: e.target.value })}
                      placeholder="e.g. Henrik Lindqvist" 
                      className="w-full max-w-md mx-auto block px-3 py-1.5 bg-white/40 border border-stone-300/40 text-xs rounded-sm focus:outline-none focus:ring-1 focus:ring-stone-400 text-center font-mono"
                    />
                  </div>

                  <div className="space-y-1 text-center">
                    <label className="text-[8px] uppercase tracking-widest font-mono text-stone-400">Electronic Address</label>
                    <input 
                      type="email" 
                      required
                      value={contactForm.email}
                      onChange={(e) => setContactForm({ ...contactForm, email: e.target.value })}
                      placeholder="e.g. henrik@lindqvist.se" 
                      className="w-full max-w-md mx-auto block px-3 py-1.5 bg-white/40 border border-stone-300/40 text-xs rounded-sm focus:outline-none focus:ring-1 focus:ring-stone-400 text-center font-mono"
                    />
                  </div>

                  <div className="space-y-1 text-center">
                    <label className="text-[8px] uppercase tracking-widest font-mono text-stone-400">Query Specifications</label>
                    <textarea 
                      required
                      rows={3}
                      value={contactForm.message}
                      onChange={(e) => setContactForm({ ...contactForm, message: e.target.value })}
                      placeholder="Explain your space, timeline, or product hopes..." 
                      className="w-full max-w-md mx-auto block px-3 py-1.5 bg-white/40 border border-stone-300/40 text-xs rounded-sm focus:outline-none focus:ring-1 focus:ring-stone-400 text-center resize-none font-mono"
                    />
                  </div>

                  <div className="text-center pt-2">
                    <button 
                      type="submit"
                      className="px-6 py-2 bg-stone-900 hover:bg-stone-950 text-white text-[9px] font-mono tracking-widest uppercase rounded-sm shadow-sm transition-all"
                    >
                      Transmit Details
                    </button>
                  </div>
                </form>
              )}
            </div>

            {/* Studio Coordinates */}
            <div className="space-y-3 text-center text-xs text-stone-500 font-mono">
              <div className="flex flex-col items-center gap-1">
                <MapPin size={12} className="text-stone-400" />
                <span className="font-semibold text-stone-800">Berlin Studio Coordinates</span>
                <span className="text-[10px] text-stone-400">Zossener Str. 11, Berlin, Germany</span>
              </div>
              <div className="flex flex-col items-center gap-1">
                <Mail size={12} className="text-stone-400" />
                <span className="font-semibold text-stone-800">Direct Inquiries</span>
                <span className="text-[10px] text-stone-400">angel@angelberger.studio</span>
              </div>
            </div>

          </div>
        </section>

      </main>

      {/* ==========================================
          MODAL: STAGE 3 EXPANDED POLAROID
          ========================================== */}
      {selectedPolaroid && (
        <div className="fixed inset-0 z-50 bg-stone-950/70 backdrop-blur-sm flex items-center justify-center p-4">
          <div className="bg-white p-4 pb-8 rounded shadow-2xl max-w-xs w-full mx-auto relative animate-scale-up border border-stone-300">
            
            <button 
              onClick={() => setSelectedPolaroid(null)}
              className="absolute top-2 right-2 p-1 text-stone-400 hover:text-stone-800 transition-colors"
              aria-label="Close Polaroid"
            >
              <X size={16} />
            </button>

            <div className="aspect-square w-full bg-stone-50 overflow-hidden shadow-inner rounded-xs mb-3">
              <img 
                src={selectedPolaroid.image} 
                alt={selectedPolaroid.title} 
                className="w-full h-full object-cover filter grayscale-[5%]" 
              />
            </div>

            <div className="text-center space-y-2">
              <h3 className="font-serif italic text-base text-stone-800">{selectedPolaroid.title}</h3>
              <div className="text-[8px] font-mono text-stone-400 uppercase">
                {selectedPolaroid.location} • {selectedPolaroid.date}
              </div>
              <div className="w-8 h-[1px] bg-stone-200 mx-auto" />
              <p className="text-[11px] text-stone-600 font-serif leading-relaxed px-1">
                {selectedPolaroid.body}
              </p>
            </div>
          </div>
        </div>
      )}

      {/* ==========================================
          MODAL: PRODUCT QUICK SPEC VIEW
          ========================================== */}
      {selectedProduct && (
        <div className="fixed inset-0 z-50 bg-stone-950/60 backdrop-blur-sm flex items-center justify-center p-4">
          <div className="bg-white rounded-lg p-5 max-w-sm w-full relative animate-scale-up shadow-2xl border border-stone-200">
            <button 
              onClick={() => setSelectedProduct(null)}
              className="absolute top-3 right-3 p-1 text-stone-400 hover:text-stone-800"
            >
              <X size={16} />
            </button>

            <div className="space-y-3">
              <div className="aspect-square bg-stone-50 overflow-hidden rounded shadow-inner">
                <img src={selectedProduct.image} alt={selectedProduct.title} className="w-full h-full object-cover" />
              </div>
              <div className="space-y-1">
                <span className="text-[8px] font-mono text-stone-400 uppercase">{selectedProduct.category}</span>
                <h3 className="font-serif text-base text-stone-800 font-semibold">{selectedProduct.title}</h3>
                <p className="text-xs text-stone-500 font-light leading-relaxed">{selectedProduct.description}</p>
              </div>
              <div className="p-2.5 bg-stone-50 rounded text-[10px] font-mono text-stone-500">
                {selectedProduct.details}
              </div>
              <div className="flex gap-2">
                <button
                  onClick={() => {
                    addToCart(selectedProduct);
                    setSelectedProduct(null);
                  }}
                  className="flex-1 py-1.5 bg-stone-950 text-[#fbf9f6] text-[9px] font-mono uppercase tracking-widest rounded-sm"
                >
                  Acquire
                </button>
                <button
                  onClick={() => setSelectedProduct(null)}
                  className="px-3 py-1.5 border border-stone-200 text-stone-600 text-[9px] font-mono uppercase"
                >
                  Back
                </button>
              </div>
            </div>
          </div>
        </div>
      )}

      {/* ==========================================
          SLIDE-OUT BASKET DRAWER
          ========================================== */}
      {isCartOpen && (
        <div className="fixed inset-0 z-50 overflow-hidden">
          <div 
            onClick={() => setIsCartOpen(false)} 
            className="absolute inset-0 bg-stone-950/40 backdrop-blur-xs transition-opacity"
          />

          <div className="absolute inset-y-0 right-0 max-w-xs w-full bg-[#fbf9f6]/95 backdrop-blur-md shadow-2xl flex flex-col justify-between border-l border-stone-300/40">
            
            <div className="p-5 border-b border-stone-200 flex items-center justify-between">
              <h3 className="font-serif text-sm text-stone-800">Your Acquisitions</h3>
              <button onClick={() => setIsCartOpen(false)} className="text-stone-400 hover:text-stone-800">
                <X size={16} />
              </button>
            </div>

            <div className="flex-1 overflow-y-auto p-5 space-y-4">
              {cart.length === 0 ? (
                <div className="text-center py-20 space-y-3">
                  <span className="text-stone-300 block text-3xl">🪹</span>
                  <p className="text-xs text-stone-400 font-mono tracking-wider">Empty Collection Drawer</p>
                </div>
              ) : (
                cart.map((item) => (
                  <div key={item.id} className="flex gap-3 border-b border-stone-200/20 pb-3 last:border-0">
                    <img src={item.image} alt={item.title} className="w-12 h-12 object-cover rounded shadow-xs shrink-0" />
                    <div className="flex-1 min-w-0 space-y-0.5">
                      <h4 className="font-serif text-xs text-stone-800 truncate">{item.title}</h4>
                      <p className="text-[10px] font-mono text-stone-500">${item.price} x {item.quantity}</p>
                      <button 
                        onClick={() => removeFromCart(item.id)}
                        className="text-[8px] font-mono text-stone-400 hover:text-rose-700 tracking-widest uppercase"
                      >
                        Remove
                      </button>
                    </div>
                  </div>
                ))
              )}
            </div>

            {cart.length > 0 && (
              <div className="p-5 border-t border-stone-200 space-y-3 bg-white/50">
                <div className="flex justify-between text-xs font-mono font-medium text-stone-800">
                  <span>Subtotal:</span>
                  <span>${cartTotal}</span>
                </div>
                <button 
                  onClick={handleCheckout}
                  className="w-full py-2 bg-stone-900 hover:bg-stone-950 text-white text-[9px] font-mono tracking-widest uppercase rounded-sm shadow-sm"
                >
                  Proceed to Secure Checkout
                </button>
              </div>
            )}
          </div>
        </div>
      )}

      {/* ==========================================
          SAFE TRANSITION STEP OVERLAY
          ========================================== */}
      {checkoutStep && (
        <div className="fixed inset-0 z-50 bg-stone-950/70 backdrop-blur-md flex items-center justify-center p-4">
          <div className="bg-white/95 border border-stone-300 p-5 max-w-xs w-full text-center rounded-lg shadow-2xl space-y-3">
            {checkoutStep === 'checking-out' ? (
              <div className="space-y-3 py-4">
                <div className="animate-spin rounded-full h-6 w-6 border-b-2 border-stone-800 mx-auto" />
                <p className="text-[9px] font-mono tracking-widest text-stone-400 uppercase">Contacting secure transactional gate...</p>
              </div>
            ) : (
              <div className="space-y-3 py-2">
                <div className="w-10 h-10 bg-stone-100 rounded-full flex items-center justify-center mx-auto border border-stone-200">
                  <Check size={16} className="text-emerald-700" />
                </div>
                <h3 className="font-serif text-base text-stone-800">Order Manifest Created</h3>
                <p className="text-xs text-stone-500 leading-relaxed">
                  A custom checkout ticket has been dispatched. Angel Berger will verify piece availability shortly.
                </p>
                <button
                  onClick={() => {
                    setCheckoutStep(null);
                    setIsCartOpen(false);
                  }}
                  className="px-4 py-1.5 bg-stone-900 text-white text-[9px] font-mono tracking-widest uppercase rounded-sm"
                >
                  Confirm & Return
                </button>
              </div>
            )}
          </div>
        </div>
      )}

    </div>
  );
}
