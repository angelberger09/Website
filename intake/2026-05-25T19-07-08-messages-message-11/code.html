export default function App() {
  const [activeSection, setActiveSection] = useState('home');
  const [starCount] = useState(45);
  const containerRef = useRef(null);

  // Home Page Interactive Frame states
  const [mousePos, setMousePos] = useState({ x: 0, y: 0 });
  const [isHeroHovered, setIsHeroHovered] = useState(false);

  // Animatic stages for Home page intro sequences
  const [animStage, setAnimStage] = useState(0); // 0: Start, 1: Move Name Up, 2: Reveal Content

  // Gallery Modal
  const [activePolaroid, setActivePolaroid] = useState(null);

  // Contact Form Feedback
  const [formState, setFormState] = useState({ name: '', email: '', message: '' });
  const [formSuccess, setFormSuccess] = useState(false);

  // Product modal / inquiry sequence
  const [inquiredProduct, setInquiredProduct] = useState(null);

  // Setup three-stage animatic sequence on mount
  useEffect(() => {
    // Stage 0 -> 1: Slide name upward after 1.2 seconds
    const timer1 = setTimeout(() => {
      setAnimStage(1);
    }, 1200);

    // Stage 1 -> 2: Reveal the secondary hero modules and tags
    const timer2 = setTimeout(() => {
      setAnimStage(2);
    }, 2000);

    return () => {
      clearTimeout(timer1);
      clearTimeout(timer2);
    };
  }, []);

  // Set up Scroll Snapping & Section tracking with IntersectionObserver
  useEffect(() => {
    const sections = ['home', 'gallery', 'blog', 'services', 'products', 'contact'];
    const observers = [];

    sections.forEach((secId) => {
      const el = document.getElementById(secId);
      if (el) {
        const observer = new IntersectionObserver(
          ([entry]) => {
            if (entry.isIntersecting) {
              setActiveSection(secId);
            }
          },
          { threshold: 0.45 }
        );
        observer.observe(el);
        observers.push({ observer, el });
      }
    });

    return () => {
      observers.forEach(({ observer, el }) => observer.unobserve(el));
    };
  }, []);

  // Handle Parallax shift for Home Atelier Frame
  const handleHeroMouseMove = (e) => {
    const rect = e.currentTarget.getBoundingClientRect();
    // Normalize coordinates to range of -0.5 to 0.5
    const x = (e.clientX - rect.left) / rect.width - 0.5;
    const y = (e.clientY - rect.top) / rect.height - 0.5;
    setMousePos({ x: x * 18, y: y * 18 }); // max offset 18px
  };

  const handleHeroMouseLeave = () => {
    setIsHeroHovered(false);
    setMousePos({ x: 0, y: 0 });
  };

  // Smooth scroll helper
  const navigateTo = (id) => {
    const el = document.getElementById(id);
    if (el) {
      el.scrollIntoView({ behavior: 'smooth' });
    }
  };

  // Form submit simulated action
  const handleContactSubmit = (e) => {
    e.preventDefault();
    if (!formState.name || !formState.email) return;
    setFormSuccess(true);
    setTimeout(() => {
      setFormSuccess(false);
      setFormState({ name: '', email: '', message: '' });
    }, 5000);
  };

  return (
    <div className="bg-[#0b0914] text-white overflow-hidden relative selection:bg-purple-900/50 selection:text-purple-200">
      <style dangerouslySetInnerHTML={{ __html: globalStyles }} />

      {/* --- BACKGROUND LAYER: STAR FIELD --- */}
      <div className="fixed inset-0 pointer-events-none z-0 overflow-hidden opacity-40">
        {[...Array(starCount)].map((_, i) => {
          const style = {
            width: `${Math.random() * 2 + 1}px`,
            height: `${Math.random() * 2 + 1}px`,
            top: `${Math.random() * 100}%`,
            left: `${Math.random() * 100}%`,
            animationDelay: `${Math.random() * 8}s`,
            animationDuration: `${25 + Math.random() * 30}s`,
            opacity: Math.random() * 0.7 + 0.1,
          };
          return <div key={i} className="star" style={style} />;
        })}
      </div>

      {/* --- BG GRADIENT SHIFT EFFECT --- */}
      <div className="fixed inset-0 pointer-events-none z-0 bg-[radial-gradient(ellipse_at_top,_var(--tw-gradient-stops))] from-purple-950/20 via-slate-950/80 to-[#050308]" />

      {/* --- VISUAL-ONLY POST-PROCESS PAPER TEXTURE OVERLAY --- */}
      <div className="paper-overlay" />

      {/* --- ANALOG SIDE SCROLLER DOT INDICATORS --- */}
      <nav className="scroll-progress hidden md:flex">
        {['home', 'gallery', 'blog', 'services', 'products', 'contact'].map((id, index) => (
          <button
            key={id}
            onClick={() => navigateTo(id)}
            className="group relative flex items-center justify-center p-2"
            aria-label={`Scroll to ${id}`}
          >
            <span className={`w-1.5 h-1.5 rounded-full transition-all duration-500 ease-out 
              ${activeSection === id ? 'bg-white scale-[1.8] shadow-lg shadow-white/30' : 'bg-white/20 group-hover:bg-white/50 scale-100'}
            `} />
            <span className="absolute left-8 px-2 py-0.5 rounded text-[10px] font-mono-tag tracking-widest text-white/40 opacity-0 group-hover:opacity-100 transition-opacity duration-300 pointer-events-none uppercase">
              {id}
            </span>
          </button>
        ))}
      </nav>

      {/* --- SYSTEM SCROLL CONTAINER --- */}
      <div className="snap-container no-scrollbar z-10 relative" ref={containerRef}>

        {/* ================= HOME SECTION ================= */}
        <section id="home" className="snap-section flex flex-col justify-between items-center py-16 px-4 md:px-12">
          {/* Top spacer */}
          <div className="h-6" />

          {/* Centered Composition Core */}
          <div className="w-full max-w-4xl flex flex-col items-center justify-center relative">
            
            {/* Stage Title Container */}
            <div className="w-full flex flex-col items-center text-center z-20 pointer-events-none mb-10">
              {/* Main Name Heading with 3-Stage Animatic upward transition */}
              <h1 className={`font-title font-bold text-5xl sm:text-7xl md:text-[90px] leading-none tracking-tight text-white select-none transition-all duration-[1200ms] ease-[cubic-bezier(0.16,1,0.3,1)]
                ${animStage === 0 ? 'translate-y-[15vh] scale-105 opacity-100' : 'translate-y-0 scale-100 opacity-100'}
              `}>
                A. BERGER
              </h1>

              {/* Subtitle with reveal */}
              <p className={`font-mono-tag text-xs md:text-sm tracking-[0.35em] text-white/60 uppercase mt-4 transition-all duration-[1000ms] ease-out delay-500
                ${animStage >= 1 ? 'opacity-100 translate-y-0' : 'opacity-0 translate-y-6'}
              `}>
                A DIVERSE CREATIVE
              </p>
            </div>

            {/* INTERACTIVE HERO ATELIER FRAME (Fade-color, unblur & parallax on hover) */}
            <div 
              onMouseMove={handleHeroMouseMove}
              onMouseLeave={handleHeroMouseLeave}
              onMouseEnter={() => setIsHeroHovered(true)}
              className={`w-full max-w-3xl aspect-[1.85] rounded-lg overflow-hidden border border-white/10 relative transition-all duration-[1200ms] ease-[cubic-bezier(0.16,1,0.3,1)] shadow-2xl cursor-crosshair
                ${animStage >= 1 ? 'opacity-100 scale-100 translate-y-0' : 'opacity-0 scale-95 translate-y-12'}
              `}
            >
              {/* The Interior Photo Layer */}
              <img 
                src="https://images.unsplash.com/photo-1513519245088-0e12902e5a38?q=80&w=1600&auto=format&fit=crop"
                alt="Angel Berger Atelier"
                className="absolute inset-0 w-full h-full object-cover transition-all duration-700 ease-[cubic-bezier(0.16,1,0.3,1)]"
                style={{
                  transform: `scale(${isHeroHovered ? 1.15 : 1.05}) translate(${mousePos.x}px, ${mousePos.y}px)`,
                  filter: isHeroHovered 
                    ? 'blur(0px) grayscale(10%) contrast(1.15)' 
                    : 'blur(5px) grayscale(65%) contrast(0.9)',
                  opacity: isHeroHovered ? 0.95 : 0.65
                }}
              />

              {/* Interactive Fade-Color Vertical Gradients */}
              <div 
                className="absolute inset-0 bg-gradient-to-t from-[#0b0914] via-black/40 to-[#0b0914] transition-opacity duration-700 pointer-events-none"
                style={{ opacity: isHeroHovered ? 0.4 : 0.75 }}
              />

              {/* Frame Label (Analog Touch) */}
              <div className="absolute bottom-4 left-6 z-20 flex items-center gap-2 pointer-events-none">
                <span className="w-1.5 h-1.5 rounded-full bg-red-500 animate-pulse" />
                <span className="font-mono-tag text-[10px] tracking-widest text-white/70 uppercase">Atelier Session No. 85 // Interactive Frame</span>
              </div>
            </div>

            {/* Resolved Stage 3 Content Blocks (Staggered slide-in) */}
            <div className={`w-full max-w-2xl mt-12 grid grid-cols-2 gap-6 transition-all duration-[1200ms] ease-out delay-700
              ${animStage === 2 ? 'opacity-100 translate-y-0' : 'opacity-0 translate-y-8'}
            `}>
              <div className="p-4 border-l border-white/10 flex flex-col justify-center">
                <span className="font-mono-tag text-[10px] text-white/40 tracking-widest uppercase mb-1">Curation</span>
                <p className="font-serif-editorial text-sm italic text-white/70">"Form follows atmosphere, materials drive the consciousness."</p>

              <div className="p-4 border-l border-white/10 flex flex-col justify-center">

          {/* Animated Scroll Indicator (Analog vibe) */}
          <div className="flex flex-col items-center mt-6 group cursor-pointer" onClick={() => navigateTo('gallery')}>
            <span className="font-mono-tag text-[10px] tracking-[0.25em] text-white/30 group-hover:text-white/60 transition-colors duration-300 uppercase mb-2">Scroll To View Archives</span>

                  <div className="mt-8 flex flex-wrap gap-1.5">
                    {service.skills.map((skill, i) => (
                      <span key={i} className="font-mono-tag text-[8px] bg-white/5 border border-white/5 text-white/40 px-2 py-0.5 rounded-full uppercase">
                        {skill}
                      </span>
                    ))}
                  </div>
                </div>
              ))}
            </div>

            {/* Meaningful visual placeholder fills */}
            <div className="mt-12 flex justify-center">
              <div className="w-full max-w-sm h-12 bg-white/[0.02] border border-white/5 rounded-full flex items-center justify-between px-6">
                <span className="font-mono-tag text-[9px] text-white/30 tracking-widest uppercase">Currently accepting Q3-Q4 Commissions</span>
                <span className="w-2 h-2 rounded-full bg-emerald-500 animate-pulse" />
              </div>
            </div>
          </div>

          <span className="font-mono-tag text-[10px] tracking-widest text-white/30 uppercase">Section 04 // Studio Practice</span>
        </section>

        {/* ================= PRODUCTS SECTION ================= */}
        <section id="products" className="snap-section flex flex-col justify-between items-center py-16 px-4 md:px-24">
          <div className="w-full max-w-4xl mt-8">
            {/* Header */}
            <div className="flex flex-col items-center text-center mb-12">
              <span className="font-mono-tag text-xs tracking-[0.2em] text-white/40 uppercase mb-2">// PHYSICAL ARTIFACTS</span>
              <h2 className="font-title text-3xl md:text-5xl tracking-tight text-white uppercase">The Store</h2>
            </div>

            {/* Product Cards Layout */}
            <div className="grid grid-cols-1 md:grid-cols-3 gap-8">
              {productList.map((prod) => (
                <div 
                  key={prod.id}
                  className="group bg-white/[0.01] border border-white/5 rounded-lg p-4 flex flex-col justify-between hover:border-white/15 transition-all duration-500"
                >
                  <div>
                    {/* Visual Showcase (Image driven first) */}
                    <div className="aspect-[5/4] w-full bg-black overflow-hidden rounded mb-4 relative">
                      <img 
                        src={prod.img} 
                        alt={prod.title} 
                        className="w-full h-full object-cover opacity-75 group-hover:opacity-100 group-hover:scale-110 transition-all duration-700"
                      />
                      <div className="absolute top-3 right-3 bg-black/60 backdrop-blur-md px-2.5 py-1 rounded text-[10px] font-mono-tag text-white/90 border border-white/10 uppercase">
                        {prod.subtitle}
                      </div>
                    </div>

                    {/* Metadata */}
                    <div className="flex justify-between items-baseline mb-2">
                      <h3 className="font-title text-base text-white/90 tracking-tight group-hover:text-white transition-colors duration-300">
                        {prod.title}
                      </h3>
                      <span className="font-mono-tag text-xs text-white/80 font-bold">{prod.price}</span>
                    </div>

                    {/* Description */}
                    <p className="font-body text-[11px] text-white/50 leading-relaxed font-light line-clamp-3">
                      {prod.desc}
                    </p>
                  </div>

                  {/* Analog Transparent Control Button */}
                  <div className="mt-6">
                    <button 
                      onClick={() => setInquiredProduct(prod)}
                      className="w-full py-2 bg-transparent hover:bg-white text-white hover:text-black border border-white/20 hover:border-white text-[11px] font-mono-tag tracking-wider uppercase rounded transition-all duration-300 active:scale-[0.98]"
                    >
                      Inquire Artifact
                    </button>
                  </div>
                </div>
              ))}
            </div>
          </div>

          <span className="font-mono-tag text-[10px] tracking-widest text-white/30 uppercase">Section 05 // Tangibles</span>
        </section>

        {/* ================= CONTACT SECTION ================= */}
        <section id="contact" className="snap-section flex flex-col justify-between items-center py-16 px-4 md:px-24">
          <div className="w-full max-w-xl mt-8">
            {/* Header */}
            <div className="flex flex-col items-center text-center mb-10">
              <span className="font-mono-tag text-xs tracking-[0.2em] text-white/40 uppercase mb-2">// ESTABLISH SIGNAL</span>
              <h2 className="font-title text-3xl md:text-5xl tracking-tight text-white uppercase">Contact</h2>
            </div>

            {/* Centered Form / Glass Panel */}
            <div className="bg-white/[0.02] backdrop-blur-md border border-white/10 p-8 rounded-2xl shadow-2xl relative overflow-hidden">
              {formSuccess ? (
                <div className="py-12 flex flex-col items-center text-center animate-fade-in">
                  <div className="w-12 h-12 rounded-full border border-emerald-500/50 flex items-center justify-center mb-6 bg-emerald-500/10">
                    <span className="text-emerald-400">✓</span>
                  </div>
                  <h3 className="font-title text-lg uppercase text-white tracking-wide">Signal Transmitted</h3>
                  <p className="font-body text-xs text-white/50 mt-2 max-w-xs">Your transmission is received. Angel Berger will verify the parameters and respond shortly.</p>
                </div>
              ) : (
                <form onSubmit={handleContactSubmit} className="space-y-5">
                  <div className="space-y-1">
                    <label className="font-mono-tag text-[10px] tracking-widest text-white/50 uppercase block">Identifier Name</label>
                    <input 
                      type="text" 
                      required
                      value={formState.name}
                      onChange={(e) => setFormState({...formState, name: e.target.value})}
                      className="w-full bg-white/[0.03] border border-white/10 rounded px-4 py-2.5 text-xs focus:outline-none focus:border-white/40 text-white transition-all duration-300" 
                      placeholder="Your Name"
                    />
                  </div>

                  <div className="space-y-1">
                    <label className="font-mono-tag text-[10px] tracking-widest text-white/50 uppercase block">Digital Address (Email)</label>
                    <input 
                      type="email" 
                      required
                      value={formState.email}
                      onChange={(e) => setFormState({...formState, email: e.target.value})}
                      className="w-full bg-white/[0.03] border border-white/10 rounded px-4 py-2.5 text-xs focus:outline-none focus:border-white/40 text-white transition-all duration-300" 
                      placeholder="Your Address"
                    />
                  </div>

                  <div className="space-y-1">
                    <label className="font-mono-tag text-[10px] tracking-widest text-white/50 uppercase block">Project Brief Outline</label>
                    <textarea 
                      rows="3"
                      value={formState.message}
                      onChange={(e) => setFormState({...formState, message: e.target.value})}
                      className="w-full bg-white/[0.03] border border-white/10 rounded px-4 py-2.5 text-xs focus:outline-none focus:border-white/40 text-white transition-all duration-300 resize-none" 
                      placeholder="Message context..."
                    />
                  </div>

                  <button 
                    type="submit"
                    className="w-full py-3 bg-white text-black font-bold font-mono-tag text-xs tracking-widest uppercase rounded shadow-lg hover:bg-neutral-200 transition-colors duration-300 active:scale-[0.98]"
                  >
                    Send Signal
                  </button>
                </form>
              )}
            </div>

            {/* Simple Direct Links Footer */}
            <div className="mt-10 flex justify-center gap-8 text-[11px] font-mono-tag tracking-wider text-white/40">
              <a href="#instagram" className="hover:text-white transition-colors">INSTAGRAM</a>
              <a href="#vimeo" className="hover:text-white transition-colors">VIMEO</a>
              <a href="#are.na" className="hover:text-white transition-colors">ARE.NA</a>
            </div>
          </div>

          <span className="font-mono-tag text-[10px] tracking-widest text-white/30 uppercase mt-2">Section 06 // Terminal</span>
        </section>

      </div>

      {/* ================= FIXED NAVIGATION BAR ================= */}
      {/* Minimalistic, Analog & Reactive Page Bar at bottom */}
      <nav className="fixed bottom-6 left-1/2 -translate-x-1/2 z-50">
        <div className="bg-black/40 backdrop-blur-xl border border-white/10 rounded-full px-5 py-2.5 flex items-center gap-2 md:gap-4 shadow-2xl">
          {[
            { id: 'home', label: 'Home' },
            { id: 'gallery', label: 'Gallery' },
            { id: 'blog', label: 'Blog' },
            { id: 'services', label: 'Services' },
            { id: 'products', label: 'Products' },
            { id: 'contact', label: 'Contact' }
          ].map((item) => {
            const isActive = activeSection === item.id;
            return (
              <button
                key={item.id}
                onClick={() => navigateTo(item.id)}
                className={`group relative px-3 py-1.5 rounded-full transition-all duration-300 flex items-center justify-center
                  ${isActive ? 'text-white' : 'text-white/45 hover:text-white/85'}
                `}
              >
                {/* Visual Active Capsule Backplate */}
                {isActive && (
                  <span className="absolute inset-0 bg-white/10 border border-white/25 rounded-full shadow-inner animate-pulse-slow pointer-events-none" />
                )}
                
                <span className="font-mono-tag text-[10px] font-bold tracking-widest uppercase relative z-10">
                  {item.label}
                </span>

                {/* Tactile indicator underline on Hover */}
                {!isActive && (
                  <span className="absolute bottom-0 w-0 h-0.5 bg-white/20 transition-all duration-300 group-hover:w-1/2" />
                )}
              </button>
            );
          })}
        </div>
      </nav>

      {/* ================= CUSTOM EXPANDED POLAROID OVERLAY ================= */}
      {activePolaroid && (
        <div 
          onClick={() => setActivePolaroid(null)}
          className="fixed inset-0 z-50 bg-[#07060ca0] backdrop-blur-md flex items-center justify-center p-4 transition-all duration-500"
        >
          <div 
            onClick={(e) => e.stopPropagation()}
            className="bg-[#f2f1ed] p-6 pb-20 rounded-sm shadow-2xl max-w-lg w-full relative border border-neutral-300 transform scale-100 transition-all duration-500 text-neutral-900"
          >
            {/* Close Button Control */}
            <button 
              onClick={() => setActivePolaroid(null)}
              className="absolute top-4 right-4 text-neutral-400 hover:text-neutral-900 font-mono-tag text-xs tracking-wider border border-neutral-200 hover:border-neutral-400 rounded-full px-2 py-1 transition-all"
              aria-label="Close Polaroid"
            >
              ✕ CLOSE
            </button>

            {/* Immersive Image Display */}
            <div className="bg-neutral-950 w-full aspect-[4/3] overflow-hidden rounded-sm relative">
              <img 
                src={activePolaroid.src} 
                alt={activePolaroid.title} 
                className="w-full h-full object-cover opacity-90"
              />
              <div className="absolute inset-0 bg-neutral-900/10 mix-blend-color-burn" />
            </div>

            {/* Structural Body Content */}
            <div className="mt-8 text-center px-4">
              <span className="font-mono-tag text-[10px] text-neutral-400 tracking-[0.25em] uppercase block mb-2">
                {activePolaroid.category}
              </span>
              <h3 className="font-serif-editorial text-2xl font-bold italic tracking-tight text-neutral-800 mb-4">
                {activePolaroid.title}
              </h3>
              <p className="font-body text-xs text-neutral-600 leading-relaxed font-light border-t border-neutral-200/60 pt-4">
                {activePolaroid.text}
              </p>
            </div>

            {/* Bottom Analog Print Tag */}
            <div className="absolute bottom-4 left-6 right-6 flex justify-between text-[8px] font-mono-tag text-neutral-400">
              <span>PRINT REF // #00{activePolaroid.id}</span>
              <span>ANGEL BERGER PHOTO ARCHIVES</span>
            </div>
          </div>
        </div>
      )}

      {/* ================= PRODUCT INQUIRY OVERLAY ================= */}
      {inquiredProduct && (
        <div 
          onClick={() => setInquiredProduct(null)}
          className="fixed inset-0 z-50 bg-[#07060ca0] backdrop-blur-md flex items-center justify-center p-4 transition-all duration-500"
        >
          <div 
            onClick={(e) => e.stopPropagation()}
            className="bg-[#0f0c1c] border border-white/10 p-8 rounded-2xl shadow-2xl max-w-md w-full relative"
          >
            <button 
              onClick={() => setInquiredProduct(null)}
              className="absolute top-4 right-4 text-white/50 hover:text-white font-mono-tag text-xs"
            >
              ✕
            </button>

            <h3 className="font-title text-xl text-white uppercase mb-1">Inquire Artifact</h3>
            <p className="font-mono-tag text-xs text-white/40 tracking-wider mb-6">{inquiredProduct.title} // {inquiredProduct.subtitle}</p>

            <div className="space-y-4">
              <p className="font-body text-xs text-white/60 leading-relaxed">
                Our physical design batches are produced in limited numbers and individually signed and packed inside our atelier. Please confirm you would like to reserve a slot for this piece.
              </p>
              <div className="border border-white/10 p-4 rounded bg-white/[0.02]">
                <span className="font-mono-tag text-[10px] text-white/30 block mb-1">Reserve Unit Value</span>
                <span className="font-title text-base text-white">{inquiredProduct.price}</span>
              </div>

              <button 
                onClick={() => {
                  setInquiredProduct(null);
                  navigateTo('contact');
                }}
                className="w-full py-3 bg-white hover:bg-neutral-200 text-black font-bold font-mono-tag text-xs tracking-widest uppercase rounded transition-colors duration-300"
              >
                Proceed to Request Form
              </button>
            </div>
          </div>
        </div>
      )}

    </div>
  );
}
