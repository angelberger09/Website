// --- BACKGROUND STARFIELD (Adapted as dark organic motes) ---
const Starfield = () => {
  const canvasRef = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    let animationFrameId;
    let stars = [];

    const init = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
      stars = Array.from({ length: 80 }, () => ({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        radius: Math.random() * 1.5 + 0.5,
        speedY: Math.random() * 0.15 + 0.05, 
        speedX: (Math.random() - 0.5) * 0.05,
        alpha: Math.random() * 0.3 + 0.1
      }));
    };

    const draw = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      stars.forEach(star => {
        ctx.fillStyle = `rgba(20, 20, 24, ${star.alpha})`; // Dark motes
        ctx.beginPath();
        ctx.arc(star.x, star.y, star.radius, 0, Math.PI * 2);
        ctx.fill();
        
        star.y += star.speedY;
        star.x += star.speedX;
        
        if (star.y > canvas.height) star.y = 0;
        if (star.x > canvas.width) star.x = 0;
        if (star.x < 0) star.x = canvas.width;
      });
      animationFrameId = requestAnimationFrame(draw);
    };

    init();
    draw();

    window.addEventListener('resize', init);
    return () => {
      cancelAnimationFrame(animationFrameId);
      window.removeEventListener('resize', init);
    };
  }, []);

  return <canvas ref={canvasRef} className="fixed inset-0 z-[-10] pointer-events-none opacity-60" />;
};

const HomeSection = ({ id, activeSection }) => {
  const [mousePos, setMousePos] = useState({ x: 0, y: 0 });
  const [isHovered, setIsHovered] = useState(false);
  const containerRef = useRef(null);

  const handleMouseMove = (e) => {
    if (!containerRef.current) return;
    const rect = containerRef.current.getBoundingClientRect();
    const x = (e.clientX - rect.left) / rect.width - 0.5;
    const y = (e.clientY - rect.top) / rect.height - 0.5;
    setMousePos({ x, y });
  };

  const handleMouseLeave = () => {
    setMousePos({ x: 0, y: 0 });
    setIsHovered(false);
  };

  const isActive = activeSection === id;

  return (
    <section 
      id={id} 
      className={`snap-start min-h-screen w-full flex flex-col items-center justify-center py-20 px-4 md:px-12 transition-all duration-700 ${isActive ? 'scene-active' : ''}`}
    >
      <div className="w-full max-w-5xl flex flex-col items-center">
        {/* Establish Stage: Sharp Light Frame */}
        <div 
          ref={containerRef}

          onMouseEnter={() => setIsHovered(true)}

// --- MAIN PORTFOLIO ROOT APPLICATION ---
export default function App() {
  const [activeSection, setActiveSection] = useState('home');
  const scrollContainerRef = useRef(null);

  const sections = [
    { id: 'home', label: 'Home' },
    { id: 'gallery', label: 'Gallery' },
    { id: 'blog', label: 'Blog' },
    { id: 'services', label: 'Services' },
    { id: 'products', label: 'Products' },
    { id: 'process', label: 'Process' },
    { id: 'contact', label: 'Contact' }
  ];

  // Intersect Observer to track scroll-snapping alignment
  useEffect(() => {
    const container = scrollContainerRef.current;
    if (!container) return;

    const observerOptions = {
      root: container,
      rootMargin: '0px',
      threshold: 0.5,
    };

    const observerCallback = (entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          setActiveSection(entry.target.id);
        }
      });
    };

    const observer = new IntersectionObserver(observerCallback, observerOptions);
    sections.forEach((sec) => {
      const el = document.getElementById(sec.id);
      if (el) observer.observe(el);
    });

    return () => observer.disconnect();
  }, []);

  const handleNavClick = (sectionId) => {
    const el = document.getElementById(sectionId);
    if (el) {
      el.scrollIntoView({ behavior: 'smooth' });
    }
  };

  return (
    <div className="relative min-h-screen bg-[#FAF9F6] text-neutral-900 font-sans overflow-hidden select-none selection:bg-neutral-900 selection:text-white">
      <GlobalStyles />
      
      {/* Decorative ambient color exposure spots - Adapted for light mode */}
      <div className="absolute top-[10%] left-[5%] w-[40vw] h-[40vw] rounded-full bg-[#E07A5F]/5 filter blur-[150px] pointer-events-none z-[-30]" />
      <div className="absolute bottom-[20%] right-[0%] w-[45vw] h-[45vw] rounded-full bg-[#81B29A]/5 filter blur-[170px] pointer-events-none z-[-30]" />
      <div className="absolute top-[60%] left-[30%] w-[35vw] h-[35vw] rounded-full bg-[#3D5A80]/5 filter blur-[140px] pointer-events-none z-[-30]" />

      <Starfield />
      
      <div className="paper-texture"></div>

      {/* Navigation - Solid light editorial style */}
      <nav className="fixed top-6 left-1/2 -translate-x-1/2 z-50 flex gap-2 md:gap-4 p-2 bg-white/90 backdrop-blur-md border border-neutral-200 shadow-[0_10px_30px_rgba(0,0,0,0.05)]">
        {sections.map(sec => (
          <button
            key={sec.id}
            onClick={() => handleNavClick(sec.id)}
            className={`px-3 py-1.5 text-[9px] md:text-[10px] uppercase tracking-[0.2em] font-bold transition-colors duration-300 relative group font-mono-custom
              ${activeSection === sec.id ? 'text-neutral-900' : 'text-neutral-400 hover:text-neutral-900'}
            `}
          >
            {sec.label}
            {/* Active Indicator Line - Stark Black */}
            <span 
              className={`absolute bottom-0 left-1/2 -translate-x-1/2 h-[2px] bg-neutral-900 transition-all duration-300 ease-out
                ${activeSection === sec.id ? 'w-[60%] opacity-100' : 'w-0 opacity-0 group-hover:w-[40%] group-hover:opacity-50'}
              `}
            />
          </button>
        ))}
      </nav>

      <main 
        ref={scrollContainerRef}
        className="absolute inset-0 z-10 overflow-y-scroll snap-y snap-mandatory h-screen w-full no-scrollbar"
      >
        <HomeSection id="home" activeSection={activeSection} />
        <GallerySection id="gallery" activeSection={activeSection} />
        <BlogSection id="blog" activeSection={activeSection} />
        <ServicesSection id="services" activeSection={activeSection} />
        <ProductsSection id="products" activeSection={activeSection} />
        <ProcessSection id="process" activeSection={activeSection} />
        <ContactSection id="contact" activeSection={activeSection} />
      </main>

    </div>
  );
}
